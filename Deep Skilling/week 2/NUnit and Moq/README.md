Assertion	Purpose
Assert.AreEqual(expected, actual)=	Checks if values are equal
Assert.AreNotEqual()	         =    Checks values are different
Assert.IsTrue(condition)	      =     Checks condition is true
Assert.IsFalse(condition)	       =   Checks condition is false
Assert.IsNull(object)	            =   Checks object is null
Assert.IsNotNull(object)	       =   Checks object is not null
Assert.Throws<Exception>()	        = Checks an exception is thrown
CollectionAssert.AreEqual()	      = Compares two collections
